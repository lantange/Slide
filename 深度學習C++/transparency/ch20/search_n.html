// 是否同為母音
bool  vowel ( char a , char b ) {
    return ( ( a == 'a' || a == 'e' || a == 'i' ||
               a == 'o' || a == 'u' ) &&
             ( b == 'a' || b == 'e' || b == 'i' ||
               b == 'o' || b == 'u' ) ? true : false ) ;
}
