// 計算字串字元數 
int  char_count( const string& s ) { 
    return s.size() ; 
}

// 將每個字母在字母表的順序旋轉字串長度個位置
string  rotate_char( string s , int no ) {

        for ( int i = 0 ; i < s.size() ; ++i ) 
            
            if ( 'a' <= s[i] && s[i] <= 'z' ) {
                s[i] +=  s.size() ;
                if ( s[i] > 'z' ) s[i] -= 26 ;
            } else if ( 'A' <= s[i] && s[i] <= 'Z' ) {
                s[i] +=  s.size() ;
                if ( s[i] > 'Z' ) s[i] -= 26 ;
            }
    
    return s ;
}
